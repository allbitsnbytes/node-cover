  /**
   * Strip the function definition from `str`,
   * and re-indent for pre whitespace.
   */
  
  function clean(str) {
    str = str
      .replace(/^function *\(.*\) *{/, '')
      .replace(/\s+\}$/, '');
  
    var spaces = str.match(/^\n?( *)/)[1].length
      , re = new RegExp('^ {' + spaces + '}', 'gm');
  
    str = str
      .replace(re, '')
      .replace(/^\s+/, '');
  
    return str;
  }
